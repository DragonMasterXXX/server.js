const express = require('express'); const app = express(); const http = require('http').Server(app); const io = require('socket.io')(http, { cors: { origin: "*" } // This allows GitHub to talk to your server }); io.on('connection', (socket) => { socket.on('join', (role) => { socket.join(role); if (role === 'viewer') { io.to('admin').emit('viewer-status', { online: true }); } }); socket.on('prank-command', (data) => { io.to('viewer').emit('exec-prank', data); }); socket.on('mouse-move-admin', (coords) => { io.to('viewer').emit('mouse-sync', coords); }); socket.on('viewer-update', (data) => { io.to('admin').emit('mirror-update', data); }); socket.on('disconnect', () => { io.to('admin').emit('viewer-status', { online: false }); }); }); const PORT = process.env.PORT || 3000; http.listen(PORT, () => console.log(`Server Live on Port ${PORT}`));
